```r
#' @keywords internal
.eco_lifestage_resolve_term <- function(org_lifestage, ecotox_release) {
  release_id <- ecotox_release

  # Alias normalization (D-03): look up before any provider call
  query_term <- .eco_lifestage_alias_lookup(org_lifestage)

  candidates <- dplyr::bind_rows(
    .eco_lifestage_query_ols4(query_term),
    .eco_lifestage_query_bioportal(query_term),
    .eco_lifestage_query_nvs(query_term),
    .eco_lifestage_query_wikidata(query_term),
    .eco_lifestage_query_agrovoc(query_term)
  )

  .eco_lifestage_rank_candidates(org_lifestage, candidates) |>
    dplyr::mutate(ecotox_release = release_id)
}
```

```r
#' @keywords internal
.eco_lifestage_auto_derive <- function(baseline, audit = NULL) {
  # Load existing derivation map as source of truth for already-curated mappings
  existing_derivation <- .eco_lifestage_derivation_map()
  category_map <- .eco_lifestage_ontology_category_map()

  # --- Part 1: Derivation for resolved terms ---
  resolved <- baseline |>
    dplyr::filter(.data$source_match_status == "resolved") |>
    dplyr::group_by(.data$org_lifestage) |>
    dplyr::slice(1) |>
    dplyr::ungroup()

  # Check existing derivation first, then fall back to category map
  resolved_keys <- resolved |>
    dplyr::distinct(.data$source_ontology, .data$source_term_id)

  # Keys already in curated derivation -> keep those
  existing_keys <- dplyr::semi_join(
    existing_derivation,
    resolved_keys,
    by = c("source_ontology", "source_term_id")
  )

  # Keys NOT in curated derivation -> derive from category map
  new_keys <- dplyr::anti_join(
    resolved_keys,
    existing_derivation,
    by = c("source_ontology", "source_term_id")
  )

  auto_derived <- new_keys |>
    dplyr::left_join(
      category_map,
      by = c("source_ontology", "source_term_id")
    ) |>
    dplyr::mutate(
      derivation_source = dplyr::if_else(
        !is.na(.data$harmonized_life_stage),
        "auto_ontology_category_map",
        "auto_unmatched_needs_review"
      ),
      harmonized_life_stage = dplyr::coalesce(
        .data$harmonized_life_stage,
        "Other/Unknown"
      ),
      reproductive_stage = dplyr::coalesce(.data$reproductive_stage, FALSE)
    )

  # --- Part 2: Derivation for unresolvable terms (from audit) ---
  unresolvable_rows <- tibble::tibble()
  if (!is.null(audit) && nrow(audit) > 0) {
    # out_of_scope_biology -> Other/Unknown with documented reason
    out_of_scope <- audit |>
      dplyr::filter(.data$triage_bucket == "out_of_scope_biology") |>
      dplyr::transmute(
        source_ontology = NA_character_,
        source_term_id = NA_character_,
        harmonized_life_stage = "Other/Unknown",
        reproductive_stage = FALSE,
        derivation_source = "auto_triage_out_of_scope"
      )

    # administrative_noise -> already handled by Phase 36.1 (4 rows in derivation CSV)
    # But ensure they exist
    admin_noise <- audit |>
      dplyr::filter(.data$triage_bucket == "administrative_noise") |>
      dplyr::transmute(
        source_ontology = NA_character_,
        source_term_id = NA_character_,
        harmonized_life_stage = "Unspecified",
        reproductive_stage = FALSE,
        derivation_source = "administrative_noise_audit"
      )

    # ambiguous terms that stayed unresolved -> Other/Unknown
    ambiguous <- audit |>
      dplyr::filter(.data$triage_bucket == "ambiguous") |>
      dplyr::transmute(
        source_ontology = NA_character_,
        source_term_id = NA_character_,
        harmonized_life_stage = "Other/Unknown",
        reproductive_stage = FALSE,
        derivation_source = "auto_triage_ambiguous"
      )

    unresolvable_rows <- dplyr::bind_rows(out_of_scope, admin_noise, ambiguous)
  }

  # Combine: existing curated + auto-derived + unresolvable
  dplyr::bind_rows(
    existing_keys,
    auto_derived,
    unresolvable_rows
  ) |>
    dplyr::distinct(.data$source_ontology, .data$source_term_id, .keep_all = TRUE) |>
    dplyr::arrange(.data$source_ontology, .data$source_term_id)
}
```

Key behaviors:
- Existing curated derivation rows are preserved as-is (source of truth)
- New resolved term IDs are auto-mapped via the ontology category map
- Terms not in the category map get `harmonized_life_stage = "Other/Unknown"` with `derivation_source = "auto_unmatched_needs_review"` for curator flagging
- Unresolvable terms (out_of_scope, admin, ambiguous) get explicit derivation rows per D-08

```r
# -- Step 4: Coverage Gate --------------------------------------------------

cli::cli_h2("Step 4: Coverage Gate (D-07)")

# Every term must either be resolved with a derivation row, OR have an
# explicit derivation row explaining why it is unresolved
baseline_terms <- unique(re_resolved$org_lifestage)
resolved_terms <- re_resolved |>

  dplyr::distinct(.data$source_ontology, .data$source_term_id)

derivation_gaps <- dplyr::anti_join(
  resolved_keys,
  new_derivation,
  by = c("source_ontology", "source_term_id")

  cli::cli_warn(c(

# Unresolved terms should have audit classification
```
